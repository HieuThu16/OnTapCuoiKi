<?php
// Xử lý thêm tài khoản
if (isset($_POST['themTaiKhoan'])) {
    $tenDangNhap = $_POST['tenDangNhap'];
    $matKhau = $_POST['matKhau'];
    $maQuyen = $_POST['maQuyen'];
    $ngayTao = date('Y-m-d');
    $tinhTrang = 0;

    // Kiểm tra tên đăng nhập đã tồn tại
    $check_stmt = $conn->prepare("SELECT maTaiKhoan FROM TAIKHOAN WHERE tenDangNhap = ?");
    $check_stmt->bind_param("s", $tenDangNhap);
    $check_stmt->execute();
    $check_result = $check_stmt->get_result();

    if ($check_result->num_rows > 0) {
        echo "<script>alert('Tên đăng nhập đã tồn tại!');</script>";
    } else {
        // Mã hóa mật khẩu
        $matKhau_hash = password_hash($matKhau, PASSWORD_DEFAULT);

        // Chuẩn bị câu lệnh INSERT
        // Sửa lại bind_param: 's' cho $ngayTao (date là string), 'i' cho $maQuyen (integer)
        $stmt = $conn->prepare("INSERT INTO TAIKHOAN (tenDangNhap, matKhau, ngayTaoTaiKhoan, tinhTrangTaiKhoan, maQuyen) VALUES (?, ?, ?, ?, ?)");
        $stmt->bind_param("ssisi", $tenDangNhap, $matKhau_hash, $ngayTao, $tinhTrang, $maQuyen);

        if ($stmt->execute()) {
            echo "<script>alert('Thêm thành công'); window.location='themtaikhoan.php';</script>";
        } else {
            echo "<script>alert('Thêm thất bại: " . $stmt->error . "');</script>"; // Thêm lỗi chi tiết để debug
        }
        $stmt->close();
    }
    $check_stmt->close();
}

$conn->close();
?>

### Chức năng xóa tài khoản

```php
<?php
// File: xoa_taikhoan.php
$host = "localhost";
$username = "root";
$password = "root";
$database = "BaiThiDB";

$conn = new mysqli($host, $username, $password, $database);
if ($conn->connect_error) { // Đảm bảo kiểm tra kết nối
    die("Kết nối thất bại: " . $conn->connect_error);
}

// Xác nhận confirm() phải được thực hiện trên client trước khi gửi request xóa
// Đoạn script confirm() dưới đây được loại bỏ vì nó không có tác dụng ngăn chặn lệnh PHP chạy.
// Thay vào đó, việc xác nhận phải được thực hiện trên trang gọi đến xoa_taikhoan.php (ví dụ: danhsach_taikhoan.php).

if (isset($_GET['id']) && is_numeric($_GET['id'])) { // Kiểm tra id tồn tại và là số hợp lệ
    $maTaiKhoan = (int)$_GET['id']; // Ép kiểu sang số nguyên để đảm bảo an toàn

    $stmt = $conn->prepare("DELETE FROM TAIKHOAN WHERE maTaiKhoan = ?");
    $stmt->bind_param("i", $maTaiKhoan);

    if ($stmt->execute()) {
        echo "<script>alert('Xóa thành công'); window.location.href='danhsach_taikhoan.php';</script>";
    } else {
        echo "<script>alert('Xóa thất bại: " . $stmt->error . "'); window.location.href='danhsach_taikhoan.php';</script>";
    }

    $stmt->close();
} else {
    // Thông báo nếu không có ID hoặc ID không hợp lệ
    echo "<script>alert('ID tài khoản không hợp lệ hoặc không được cung cấp!'); window.location.href='danhsach_taikhoan.php';</script>";
}

$conn->close();
?>
```
